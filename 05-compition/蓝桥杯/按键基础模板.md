``` C
typedef struct {
    uint8_t judge_sta;      // 状态机变量
    bool short_flag;        // 短按标志位
    bool long_flag;         // 长按标志位
    uint32_t press_tick;    // 按下时刻计数
} Key_Struct;

Key_Struct Keys[4]; // 对应 B1, B2, B3, B4

void Key_Scan(void) {
    // 读取 4 个按键引脚状态 (假设按下为低电平)
    uint8_t key_val[4];
    key_val[0] = HAL_GPIO_ReadPin(GPIOB, GPIO_PIN_0);
    key_val[1] = HAL_GPIO_ReadPin(GPIOB, GPIO_PIN_1);
    key_val[2] = HAL_GPIO_ReadPin(GPIOB, GPIO_PIN_2);
    key_val[3] = HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_0);

    for (int i = 0; i < 4; i++) {
        switch (Keys[i].judge_sta) {
            case 0: // 状态 0：判断是否按下
                if (key_val[i] == 0) {
                    Keys[i].judge_sta = 1; // 转入消抖
                    Keys[i].press_tick = HAL_GetTick(); // 记录按下起始时间
                }
                break;

            case 1: // 状态 1：消抖与确认
                if (key_val[i] == 1) {
                    Keys[i].judge_sta = 0; // 抖动，返回
                } else if (HAL_GetTick() - Keys[i].press_tick >= 10) { 
                    Keys[i].judge_sta = 2; // 确认按下，转入长短按判定
                }
                break;

            case 2: // 状态 2：等待松手
                if (key_val[i] == 1) { // 此时松手了
                    if (HAL_GetTick() - Keys[i].press_tick < 800) {
                        Keys[i].short_flag = true; // 短按确认
                    }
                    Keys[i].judge_sta = 0; // 重置
                } else if (HAL_GetTick() - Keys[i].press_tick >= 800) {
                    Keys[i].long_flag = true;  // 达到 800ms，判定为长按
                    Keys[i].judge_sta = 3;     // 转到状态 3 等待最后松手
                }
                break;

            case 3: // 状态 3：长按后的死区，等待按键抬起
                if (key_val[i] == 1) Keys[i].judge_sta = 0;
                break;
        }
    }
}
```
CubeMX设置(GPIO_Input)上拉模式（pull-up),按下为0，抬起为1
