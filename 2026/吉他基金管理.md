```dataview
TABLE 
    without ID
    L.text AS "已完成任务", 
    L.金额 AS "奖励"

FLATTEN file.lists AS L
WHERE L.completed = true AND L.金额 > 0
```


```dataviewjs
let target = 3000; // 你的吉他目标价格
let pages = dv.pages().file.lists.where(l => l.completed && l.金额 > 0);
let total = pages.金额.array().reduce((a, b) => a + b, 0);
let progress = ((total / target) * 100).toFixed(2);

dv.header(2, "🎸 吉他基金总额: " + total + " 元");
dv.paragraph("当前进度: " + progress + "%");
dv.paragraph("<progress value='" + total + "' max='" + target + "'></progress>");

if (total >= target) {
    dv.paragraph("🎉 恭喜！你可以去买吉他了！");
} else {
    dv.paragraph("加油！还差 " + (target - total) + " 元。");
}
```
