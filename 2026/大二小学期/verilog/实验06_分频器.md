# 实验 06：分频器与时钟生成

## 实验目标

- 掌握分频器原理（偶数分频 / 奇数分频）
- 实现 50% 占空比的时钟输出
- 用分频后的时钟驱动 LED

## 代码实现

### 偶数分频

```verilog
// 50MHz → 1Hz (50_000_000 分频)
module clk_div_even #(
    parameter DIV_N = 50_000_000
) (
    input  clk_in, rst_n,
    output reg clk_out
);
    integer cnt;

    always @(posedge clk_in or negedge rst_n) begin
        if (!rst_n) begin
            cnt     <= 0;
            clk_out <= 1'b0;
        end else begin
            if (cnt == (DIV_N/2 - 1)) begin
                cnt     <= 0;
                clk_out <= ~clk_out;
            end else begin
                cnt <= cnt + 1;
            end
        end
    end
endmodule
```

### 奇数分频

```verilog
// 5 分频（占空比 50%）
module clk_div_odd #(
    parameter DIV_N = 5
) (
    input  clk_in, rst_n,
    output clk_out
);
    reg [3:0] cnt1, cnt2;
    reg clk1, clk2;

    // 上升沿计数
    always @(posedge clk_in or negedge rst_n) begin
        if (!rst_n) begin
            cnt1 <= 0;
            clk1 <= 0;
        end else begin
            if (cnt1 == DIV_N-1)
                cnt1 <= 0;
            else
                cnt1 <= cnt1 + 1;
            clk1 <= (cnt1 < DIV_N/2) ? 1 : 0;
        end
    end

    // 下降沿计数
    always @(negedge clk_in or negedge rst_n) begin
        if (!rst_n) begin
            cnt2 <= 0;
            clk2 <= 0;
        end else begin
            if (cnt2 == DIV_N-1)
                cnt2 <= 0;
            else
                cnt2 <= cnt2 + 1;
            clk2 <= (cnt2 < DIV_N/2) ? 1 : 0;
        end
    end

    assign clk_out = clk1 | clk2;  // 相或得 50% 占空比
endmodule
```
